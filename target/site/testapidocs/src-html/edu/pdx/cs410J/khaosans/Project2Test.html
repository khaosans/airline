036                "args are (in this order):\n" +
037                "name The name of the airline\n" +
038                "flightNumber The flight number\n" +
039                "src Three-letter code of departure airport\n" +
040                "departTime Departure date and time (24-hour time)\n" +
041                "dest Three-letter code of arrival airport\n" +
042                "arriveTime Arrival date and time (24-hour time)\n" +
